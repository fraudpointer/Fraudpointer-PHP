Interface IClient

Description
Description | Methods (details)

Fraudpointer.API.IClient is the interface that each PHP client should implement in order to be used as a client proxy to access FraudPointer Service.
Important Note: Fraudpointer PHP client library comes with a ready to use implementation of this service. You need to use Fraudpointer.API.ClientFactory to instantiate this implementation and start using FraudPointer Service.

As a user of the FraudPointer Client Library you should declare a variable of IClient type and then use ClientFactory::Construct() to get an instance of a class that implements this interface.

Below, you can see a sample client that uses IClient to communicate with FraudPointer Server:

  1.  function BaseUrl () {
  2.     return "https://production.fraudpointer.com/api/assessment_sessions";
  3.  } // BaseUrl ()
  4.  //--------------
  5.  
  6.  function ApiKey () {
  7.   return "45440beccd884ddsa688asdfasda18b09";
  8.  } // ApiKey ()
  9.  //-------------
  10.  
  11.  function Main()
  12.  {
  13.         // Instantiate the client object that implements the IClient interface
  14.         //
  15.         $client = \FraudPointer\API\ClientFactory::Construct(BaseUrl(), ApiKey());
  16.  
  17.         // 1st STEP: Create Assessment Session. This goes to the FraudPointer Server and gets back an
  18.         //           Assessment Session object
  19.         //
  20.         $assessment_session = $client->CreateAssessmentSession();
  21.  
  22.         // 2nd STEP: Create an Event object of "Generic" Type.
  23.         //
  24.         $event = new \Fraudpointer\API\Models\Event(\Fraudpointer\API\Models\Event::$GENERIC_EVENT);
  25.  
  26.         // 3rd STEP: Add data to the created Event
  27.  
  28.         $event.AddData("BookingCode", "B2YB32");
  29.         $event.AddData("StartDate", "20110322");
  30.         $event.AddData("ReservationDate", "20110321");
  31.  
  32.         // 4th STEP: Send Event to FraudPointer Server to store it in the database
  33.         //           alongside with the already stored AssessmentSession
  34.         //
  35.         $client.AppendEventToAssessmentSession($assessmentSession, $event);
  36.  
  37.         // Repeat step 3 and 4. Note that you can create as many Events as you like for the
  38.         //                      same Assessment Session
  39.         //
  40.         $event = new \Fraudpointer\API\Models\Event(\Fraudpointer\API\Models\Event::$FAILED_PAYMENT);
  41.  
  42.         $event.AddDateData("AttemptDate", time());
  43.  
  44.         $event.AddData("BILLING_ADDRESS_STREE_NAME", "Othonos");
  45.  
  46.         $client.AppendEventToAssessmentSession($assessmentSession, $event);
  47.  
  48.         // 5th STEP: Ask FraudPointer Server to carry out a Fraud Assessment. This is the real value
  49.         //           of the Service because it can tell you real time about the fraud risk of the
  50.         //           transaction that you are trying to handle.
  51.         //
  52.         $fa = $client.CreateFraudAssessment($assessmentSession, false);
  53.         return $fa.score;
  54.    }
