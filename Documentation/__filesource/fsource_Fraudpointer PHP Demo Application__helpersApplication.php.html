  121. function CreateAndSendCheckoutEvent($client, $assessment_session) {
  122.     
  123.     // Create a new Event Instance
  124.     $event = new \Fraudpointer\API\Models\Event(\Fraudpointer\API\Models\Event::$CHECKOUT_EVENT);
  125.     
  126.     // ACME_DEPARTURE_CITY is an Account Custom Session Attribute defined in Fraud Pointer Application
  127.     $event->AddData("ACME_DEPARTURE_CITY", $cities_from[$_POST["select_city_from"]]);
  128.     
  129.     // ACME_ARRIVAL_CITY is an Account Custome Session Attribute defined in Fraud Pointer Application
  130.     $event->AddData("ACME_ARRIVAL_CITY", $cities_to[$_POST["select_city_to"]]);
  131.     
  132.     // PURCHASE_AMOUNT is a System Session Attribute
  133.     $event->AddDecimalData("PURCHASE_AMOUNT", CalculatePrice($_POST["select_city_from"], $_POST["select_city_to"]));
  134.     
  135.     // CC_HASH is a System Attribute. You should always use the CreditCardHash to hash your 
  136.     // Credit Card Number and send it to Fraud Pointer Server.
  137.     $event->AddData("CC_HASH", $client->CreditCardHash($_POST["txtbx_credit_card_number"]));            
  138.     
  139.     // CC_CARD_HOLDER_NAME is a System Attribute.
  140.     $event->AddData("CC_CARD_HOLDER_NAME", $_POST["txtbx_card_holder_name"]);
  141.     
  142.     // CC_BANK_NAME is a System Attribute
  143.     $event->AddData("CC_BANK_NAME", $_POST["txtbx_bank_name_of_card"]);
  144.     
  145.     // CREDIT_CARD_FIRST_6_DIGITS
  146.     $event->AddData("CREDIT_CARD_FIRST_6_DIGITS", substr($_POST["txtbx_credit_card_number"], 0, 6));
  147.     
  148.     // E_TRAVEL_SA_PURCHASE_DATE    
  149.     $event->AddDateValue("PURCHASE_DATE", time()); 
  150.         
  151.     try {        
  152.         // having populated the Event object, we will send it to FraudPointer Server
  153.         $event_returned = $client->AppendEventToAssessmentSession($assessment_session, $event);
  154.         
  155.         return $event_returned;
  156.         
  157.     }
  158.     catch (\Fraudpointer\API\ClientException $e) {
  159.         return $e->getMessage();    
  160.     }
  161.     
  162. } // CreateAndSendCheckoutEvent ()
